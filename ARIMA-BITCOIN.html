1. Preparacin de la serie de tiempo

attach(datat)

# Especificar el periodo de tiempo y convertir la base a serie de tiempo
arima.ts <- ts(datat$BTC, start = c(2015, 1), frequency = 12)

# Transformar a log para reducir dispersin
larima.ts <- log(arima.ts)

# Verificar inicio y fin de la serie
start(larima.ts); end(larima.ts)
